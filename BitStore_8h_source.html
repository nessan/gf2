70concept BitStore = requires(Store store, const Store const_store) {
72 typename Store::word_type;
73 requires Unsigned<typename Store::word_type>;

76 { const_store.size() } -> std::same_as<usize>;

94
105 { const_store.word(usize{}) } -> std::same_as<typename Store::word_type>;

295template<BitStore Store>
296constexpr auto&
297swap(Store& store, usize i0, usize i1) {
298 gf2_debug_assert(i0 < store.size(), "index {} is out of bounds for a store of length {}", i0, store.size());
299 gf2_debug_assert(i1 < store.size(), "index {} is out of bounds for a store of length {}", i1, store.size());
300 if (i0 != i1) {
301 auto [w0, mask0] = index_and_mask<typename Store::word_type>(i0);
302 auto [w1, mask1] = index_and_mask<typename Store::word_type>(i1);
303 auto word0 = store.word(w0);
304 auto word1 = store.word(w1);
305 auto val0 = (word0 & mask0) != 0;
306 auto val1 = (word1 & mask1) != 0;
307 if (val0 != val1) {
308 if (w0 == w1) {
309 // Both bits are in the same word
310 store.set_word(w0, word0 ^ mask0 ^ mask1);
311 } else {
312 // The two bits are in different words
313 store.set_word(w0, word0 ^ mask0);
314 store.set_word(w1, word1 ^ mask1);
315 }
316 }
317 }
318 return store;
319}

372template<BitStore Store>
373constexpr bool
374all(Store const& store) {
375 auto num_words = store.words();
376 if (num_words == 0) { return true; }
377
378 // Check the fully occupied words ...
379 using word_type = typename Store::word_type;
380 for (auto i = 0uz; i < num_words - 1; ++i) {
381 if (store.word(i) != MAX<word_type>) return false;
382 }
383
384 // Last word might not be fully occupied ...
385 auto bits_per_word = BITS<word_type>;
386 auto tail_bits = store.size() % bits_per_word;
387 auto unused_bits = tail_bits == 0 ? 0 : bits_per_word - tail_bits;
388 auto last_word_max = MAX<word_type> >> unused_bits;
389 if (store.word(num_words - 1) != last_word_max) return false;
390
391 return true;
392}

471template<Unsigned Src, BitStore Store>
472constexpr auto&
473copy(Src src, Store& store) {
474 constexpr auto src_bits = BITS<Src>;
475 gf2_assert(store.size() == src_bits, "Lengths do not match: {} != {}.", store.size(), src_bits);
476
477 using word_type = typename Store::word_type;
478 constexpr auto bits_per_word = BITS<word_type>;
479 if constexpr (src_bits <= bits_per_word) {
480 // The source fits into a single store word
481 store.set_word(0, static_cast<word_type>(src));
482 } else {
483 // The source spans some number of store words (which we assume is an integer number).
484 constexpr auto num_words = src_bits / bits_per_word;
485 for (auto i = 0uz; i < num_words; ++i) {
486 auto shift = i * bits_per_word;
487 auto word_value = static_cast<word_type>((src >> shift) & MAX<word_type>);
488 store.set_word(i, word_value);
489 }
490 }
491 return store;
492}

508template<BitStore Src, BitStore Store>
509constexpr auto&
510copy(Src const& src, Store& store) {
511 // The sizes must match ...
512 gf2_assert(store.size() == src.size(), "Lengths do not match: {} != {}.", store.size(), src.size());
513
514 // What is the word types (without any const qualifiers)?
515 using word_type = std::remove_const_t<typename Store::word_type>;
516 using src_word_type = std::remove_const_t<typename Src::word_type>;
517
518 // Numbers of bits per respective words
519 constexpr auto word_bits = BITS<word_type>;
520 constexpr auto src_word_bits = BITS<src_word_type>;
521
522 // What we do next depends on the size of the respective words.
523 if constexpr (word_bits == src_word_bits) {
524 // Easiest case: Same sized words -- we can copy whole words directly ...
525 for (auto i = 0uz; i < store.words(); ++i) {
526 auto src_word = static_cast<word_type>(src.word(i));
527 store.set_word(i, src_word);
528 }
529 } else if constexpr (word_bits > src_word_bits) {
530 // Our word type is larger -- some number of source words fit into each store word.
531 // We assume that our word size is an integer multiple of the source word size.
532 constexpr auto ratio = word_bits / src_word_bits;
533
534 auto src_words = src.words();
535 for (auto i = 0uz; i < store.words(); ++i) {
536
537 // Combine `ratio` source words into a single destination word being careful not to run off the end ...
538 word_type dst_word = 0;
539 for (auto j = 0uz; j < ratio; ++j) {
540 auto src_index = i * ratio + j;
541 word_type src_word = 0;
542 if (src_index < src_words) src_word = static_cast<word_type>(src.word(src_index));
543 dst_word |= (src_word << (j * src_word_bits));
544 }
545 store.set_word(i, dst_word);
546 }
547 } else {
548 // The store word size is smaller -- each source word becomes multiple destination words.
549 // We assume that the source word size is an integer multiple of our word size.
550 constexpr auto ratio = src_word_bits / word_bits;
551
552 auto dst_words = store.words();
553 for (auto src_index = 0uz; src_index < src.words(); ++src_index) {
554 auto src_word = src.word(src_index);
555
556 // Split the source word into `ratio` destination words ...
557 for (auto j = 0uz; j < ratio; ++j) {
558 auto dst_index = src_index * ratio + j;
559 if (dst_index >= dst_words) break;
560 auto shift = j * word_bits;
561 auto dst_word = static_cast<word_type>((src_word >> shift) & MAX<word_type>);
562 store.set_word(dst_index, dst_word);
563 }
564 }
565 }
566 return store;
567}

628template<BitStore Store>
629constexpr auto&
630fill_random(Store& store, double p = 0.5, u64 seed = 0) {
631 // Keep a single static RNG per thread for all calls to this method, seeded with entropy on the first call.
632 thread_local RNG rng;
633
634 // Edge case handling ...
635 if (p < 0) return set_all(store, false);
636
637 // Scale p by 2^64 to remove floating point arithmetic from the main loop below.
638 // If we determine p rounds to 1 then we can just set all elements to 1 and return early.
639 p = p * 0x1p64 + 0.5;
640 if (p >= 0x1p64) return set_all(store);
641
642 // p does not round to 1 so we use a 64-bit URNG and check each draw against the 64-bit scaled p.
643 auto scaled_p = static_cast<u64>(p);
644
645 // If a seed was provided, set the RNG's seed to it. Otherwise, we carry on from where we left off.
646 u64 old_seed = rng.seed();
647 if (seed != 0) rng.set_seed(seed);
648
649 set_all(store, false);
650 for (auto i = 0uz; i < store.size(); ++i)
651 if (rng.u64() < scaled_p) set(store, i);
652
653 // Restore the old seed if necessary.
654 if (seed != 0) rng.set_seed(old_seed);
655 return store;
656}

705template<BitStore Store>
706constexpr usize
707leading_zeros(Store const& store) {
708 // Note: Even if the last word is not fully occupied, we know unused bits are set to 0.
709 auto bits_per_word = BITS<typename Store::word_type>;
710 for (auto i = 0uz; i < store.words(); ++i) {
711 auto w = store.word(i);
712 if (w != 0) return i * bits_per_word + static_cast<usize>(gf2::trailing_zeros(w));
713 }
714 return store.size();
715}

726template<BitStore Store>
727constexpr usize
728trailing_zeros(Store const& store) {
729 if (is_empty(store)) return 0;
730
731 // The last word may not be fully occupied so we need to subtract the number of unused bits.
732 auto bits_per_word = BITS<typename Store::word_type>;
733 auto tail_bits = store.size() % bits_per_word;
734 auto unused_bits = tail_bits == 0 ? 0 : bits_per_word - tail_bits;
735 auto num_words = store.words();
736 auto i = num_words;
737 while (i--) {
738 auto w = store.word(i);
739 if (w != 0) {
740 auto whole_count = (num_words - i - 1) * bits_per_word;
741 auto partial_count = static_cast<usize>(gf2::leading_zeros(w)) - unused_bits;
742 return whole_count + partial_count;
743 }
744 }
745 return store.size();
746}

816template<BitStore Store>
817constexpr std::optional<usize>
818next_set(Store const& store, usize index) {
819 // Start our search at index + 1.
820 index = index + 1;
821
822 // Perhaps we are off the end? (This also handles the case of an empty type).
823 if (index >= store.size()) return {};
824
825 // Where is that starting index located in the word store?
826 using word_type = typename Store::word_type;
827 auto bits_per_word = BITS<word_type>;
828 auto [word_index, bit] = index_and_offset<word_type>(index);
829
830 // Iterate forward looking for a word with a new set bit and use the lowest one ...
831 for (auto i = word_index; i < store.words(); ++i) {
832 auto w = store.word(i);
833 if (i == word_index) {
834 // First word -- turn all the bits before our starting bit into zeros so we don't see them as set.
835 reset_bits(w, 0, bit);
836 }
837 if (auto loc = lowest_set_bit(w)) return i * bits_per_word + loc.value();
838 }
839 return {};
840}

1173template<BitStore Store>
1174static constexpr auto
1175span(Store const& store, usize begin, usize end) {
1176 // Check that the range is correctly formed & does not extend beyond the end of the bit-store.
1177 gf2_assert(begin <= end, "Span range [{}, {}) is mis-ordered.", begin, end);
1178 gf2_assert(end <= store.size(), "Span end {} extends beyond the store end {}.", end, store.size());
1179
1180 // Get the zero-based word index and bit offset in that word for the begin location.
1181 using word_type = typename Store::word_type;
1182 auto bits_per_word = BITS<word_type>;
1183 auto [data_index, bit_offset] = index_and_offset<word_type>(begin);
1184
1185 // If the store is already a span, we need to adjust those values.
1186 bit_offset += store.offset();
1187 if (bit_offset >= bits_per_word) {
1188 data_index += 1;
1189 bit_offset %= bits_per_word;
1190 }
1191
1192 // Return the read-only bit-span over the underlying word data.
1193 return BitSpan<const word_type>(store.store() + data_index, bit_offset, end - begin);
1194}

1218 using word_type = typename Store::word_type;

1360template<BitStore Store>
1361constexpr void
1362riffle(Store const& store, BitVector<typename Store::word_type>& dst) {
1363 // Nothing to do if the store is empty or has just one element. With two bits `ab` we fill `dst` with `a0b`.
1364 auto sz = store.size();
1365 if (sz < 2) {
1366 dst.resize(sz);
1367 dst.copy(store);
1368 return;
1369 }
1370
1371 // Make sure `dst` is large enough to hold the riffled bits (a bit too big but we fix that below).
1372 dst.resize(2 * sz);
1373 auto dst_words = dst.words();
1374
1375 // Riffle each word in the store into two adjacent words in `dst`.
1376 for (auto i = 0uz; i < store.words(); ++i) {
1377 auto [lo, hi] = riffle(store.word(i));
1378 dst.set_word(2 * i, lo);
1379
1380 // Note that `hi` may be completely superfluous ...
1381 if (2 * i + 1 < dst_words) dst.set_word(2 * i + 1, hi);
1382 }
1383
1384 // If this bit-store was say `abcde` then `dst` will now be `a0b0c0d0e0`. Pop the last 0.
1385 dst.pop();
1386}

1429template<BitStore Store>
1430static std::string
1431to_binary_string(Store const& store, std::string_view sep = "", std::string_view pre = "", std::string_view post = "") {
1432 // Edge case ...
1433 if (is_empty(store)) { return std::format("{}{}", pre, post); }
1434
1435 // Start with the raw binary string by iterating over the words ...
1436 // We preallocate enough space to hold the entire string (actually this is usually a bit bigger than we need).
1437 auto num_words = store.words();
1438 std::string binary_string;
1439 binary_string.reserve(num_words * BITS<typename Store::word_type>);
1440 for (auto i = 0uz; i < num_words; ++i) {
1441 auto w = reverse_bits(store.word(i));
1442 binary_string.append(gf2::to_binary_string(w));
1443 }
1444
1445 // The last word may not be fully occupied and we need to remove the spurious zeros
1446 binary_string.resize(store.size());
1447
1448 // If there is no separator, return early ...
1449 if (sep.empty()) { return std::format("{}{}{}", pre, binary_string, post); }
1450
1451 // Build `result` with separators between each character ...
1452 std::string result;
1453 result.reserve(pre.size() + store.size() * (sep.size() + 1) + post.size());
1454 result.append(pre);
1455 for (auto i = 0uz; i < store.size(); ++i) {
1456 result.push_back(binary_string[i]);
1457 if (i != store.size() - 1) { result.append(sep); }
1458 }
1459 result.append(post);
1460 return result;

1534template<BitStore Store>
1535static std::string
1536to_hex_string(Store const& store) {
1537 // Edge case: No bits in the store, return the empty string.
1538 if (is_empty(store)) return "";
1539
1540 // The number of digits in the output string. Generally hexadecimal but the last may be to a lower base.
1541 auto digits = (store.size() + 3) / 4;
1542
1543 // Preallocate space allowing for a possible lower base on the last digit such as "_2".
1544 std::string result;
1545 result.reserve(digits + 2);
1546
1547 // Reverse the bits in each word to vector-order and get the fully padded hex string representation.
1548 using word_type = typename Store::word_type;
1549 for (auto i = 0uz; i < store.words(); ++i) {
1550 auto w = reverse_bits(store.word(i));
1551 result.append(gf2::to_hex_string<word_type>(w));
1552 }
1553
1554 // Last word may not be fully occupied and padded with spurious zeros so we truncate the output string.
1555 result.resize(digits);
1556
1557 // Every four elements is encoded by a single hex digit but `size()` may not be a multiple of 4.
1558 auto k = store.size() % 4;
1559 if (k != 0) {
1560 // That last hex digit should really be encoded to a lower base -- 2, 4 or 8.
1561 // We compute the number represented by the trailing `k` elements in the bit-vector.
1562 int num = 0;
1563 for (auto i = 0uz; i < k; ++i)
1564 if (get(store, store.size() - 1 - i)) num |= 1 << i;
1565
1566 // Convert that number to hex & use it to *replace* the last hex digit in our `result` string.
1567 // Also append the appropriate base to the output string so that the last digit can be interpreted properly.
1568 result.resize(result.size() - 1);
1569 result.append(std::format("{:X}.{}", num, 1 << k));
1570 }
1571 return result;
1572}

1654template<BitStore Store>
1655constexpr void
1656operator<<=(Store& store, usize shift) {
1657 // Edge cases where there is nothing to do.
1658 if (shift == 0 || store.size() == 0) return;
1659
1660 // Perhaps we are shifting all the bits out and are left with all zeros.
1661 if (shift >= store.size()) {
1662 set_all(store, false);
1663 return;
1664 }
1665
1666 // For larger shifts, we can efficiently shift by whole words first.
1667 using word_type = typename Store::word_type;
1668 auto bits_per_word = BITS<word_type>;
1669 usize word_shift = shift / bits_per_word;
1670 usize end_word = store.words() - word_shift;
1671
1672 // Do the whole word shifts first, pushing in zero words to fill the empty slots.
1673 if (word_shift > 0) {
1674 // Shift whole words -- starting at the beginning of the store.
1675 for (auto i = 0uz; i < end_word; ++i) store.set_word(i, store.word(i + word_shift));
1676
1677 // Fill in the high order words with zeros.
1678 for (auto i = end_word; i < store.words(); ++i) store.set_word(i, 0);
1679
1680 // How many bits are left to shift?
1681 shift -= word_shift * bits_per_word;
1682 }
1683
1684 // Perhaps there are some partial word shifts left to do.
1685 if (shift != 0) {
1686 // Do the "interior" words where the shift moves bits from one word to the next.
1687 if (end_word > 0) {
1688 auto shift_complement = bits_per_word - shift;
1689 for (auto i = 0uz; i < end_word - 1; ++i) {
1690 auto lo = static_cast<word_type>(store.word(i) >> shift);
1691 auto hi = static_cast<word_type>(store.word(i + 1) << shift_complement);
1692 store.set_word(i, lo | hi);
1693 }
1694 }
1695
1696 // Do the last word.
1697 auto value = store.word(end_word - 1);
1698 store.set_word(end_word - 1, static_cast<word_type>(value >> shift));
1699 }

2110convolve(Lhs const& lhs, Rhs const& rhs) {
2111 using word_type = typename Lhs::word_type;
2112 auto bits_per_word = BITS<word_type>;
2113
2114 // Edge case: if either store is empty then the convolution is empty.
2115 if (lhs.is_empty() || rhs.is_empty()) return BitVector<word_type>{};
2116
2117 // Set up the result bit-vector.
2118 auto result = BitVector<word_type>::zeros(lhs.size() + rhs.size() - 1);

2121 if (lhs.none() || rhs.none()) return result;

2124 // We have already checked that `rhs` is not all zeros so we know there is a last set bit!

2126
2127 // Initialize `result` by copying the live words from `rhs`
2128 for (auto i = 0uz; i < rhs_words_end; ++i) result.set_word(i, rhs.word(i));

2131 for (auto i = lhs.last_set().value(); i-- > 0;) {
2132 word_type prev = 0;

2134 auto left = static_cast<word_type>(prev >> (bits_per_word - 1));
2135 prev = result.word(j);
2136 result.set_word(j, static_cast<word_type>(prev << 1) | left);

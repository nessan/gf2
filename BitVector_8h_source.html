411 static std::optional<BitVector> from_string(std::string_view sv) {
412 // Edge case ...
413 if (sv.empty()) return BitVector<Word>{};
414
415 // Remove any whitespace, commas, single quotes, or underscores characters.
416 std::string s{sv};
417 std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '\'' || c == '_'; });
418 bool no_punctuation = true;
419
420 // Check for a binary prefix.
421 if (s.starts_with("0b")) return from_binary_string(s, no_punctuation);
422
423 // Check for a hex prefix.
424 if (s.starts_with("0x") || s.starts_with("0X")) return from_hex_string(s, no_punctuation);
425
426 // No prefix, but perhaps the string only contains '0' and '1' characters.
427 if (std::ranges::all_of(s, [](char c) { return c == '0' || c == '1'; })) {
428 return BitVector::from_binary_string(s, no_punctuation);
429 }
430
431 // Last gasp -- try hex ...
432 return BitVector::from_hex_string(s, no_punctuation);
433 }

448 static std::optional<BitVector> from_binary_string(std::string_view sv, bool no_punctuation = false) {
449 // Edge case ...
450 if (sv.empty()) return BitVector<Word>{};
451
452 // Convert to a string & remove the "0b" prefix if it exists.
453 std::string s{sv};
454 if (s.starts_with("0b")) s = s.substr(2);
455
456 // If necessary, also remove any whitespace, commas, single quotes, or underscores.
457 if (!no_punctuation) std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '\'' || c == '_'; });
458
459 // The string should now be a sequence of 0's and 1's.
460 if (std::ranges::any_of(s, [](char c) { return c != '0' && c != '1'; })) return std::nullopt;
461
462 // Construct the bit-vector.
463 auto result = BitVector::zeros(s.size());
464 for (auto i = 0uz; i < s.size(); ++i)
465 if (s[i] == '1') result.set(i);
466 return result;
467 }

489 static std::optional<BitVector> from_hex_string(std::string_view sv, bool no_punctuation = false) {
490 // Edge case ...
491 if (sv.empty()) return BitVector<Word>{};
492
493 // Convert to a string and remove the optional "0x" prefix if it exists.
494 std::string s{sv};
495 if (s.starts_with("0x") || s.starts_with("0X")) s = s.substr(2);
496
497 // By default, the base of the last digit is 16 just like all the others.
498 // However, there may be a suffix of the form ".base" where `base` is one of 2, 4 or 8.
499 int last_digit_base = 16;
500 if (s.ends_with(".2"))
501 last_digit_base = 2;
502 else if (s.ends_with(".4"))
503 last_digit_base = 4;
504 else if (s.ends_with(".8"))
505 last_digit_base = 8;
506
507 // Remove the suffix if it exists.
508 if (last_digit_base != 16) s = s.substr(0, s.size() - 2);
509
510 // If necessary, also remove any whitespace, commas, single quotes, or underscores.
511 if (!no_punctuation) std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '_'; });
512
513 // The string should now be a sequence of 0-9, A-F, a-f.
514 if (std::ranges::any_of(s, [](char c) { return !std::isxdigit(c); })) return std::nullopt;
515
516 // Construct the bit-vector where we allow all the characters to be hex digits.
517 auto result = BitVector::with_capacity(s.size() * 4);
518
519 // Push all but the last character -- those are hex digits for sure.
520 for (auto i = 0uz; i < s.size() - 1; ++i) result.append_hex_digit(s[i]);
521
522 // Push the last character -- it may be a hex digit or a base indicator.
523 result.append_digit(s[s.size() - 1], last_digit_base);
524
525 return result;
526 }

594 constexpr BitVector& resize(usize n) {
595 if (n != size()) {
596 m_store.resize(gf2::words_needed<Word>(n), 0);
597 auto old_size = size();
598 m_size = n;

600 // If we truncated, we need to clean up the last occupied word if necessary.
601 if (n < old_size) clean();
602 }

609 constexpr void clean() {
610 // NOTE: This works fine even if `size() == 0`.
611 auto shift = m_size % bits_per_word;
612 if (shift != 0) m_store[words() - 1] &= Word(~(MAX<Word> << shift));
613 }

746 constexpr BitVector& append_digit(char c, int base) {
747 // The known bases are 2, 4, 8, and 16.
748 static constexpr std::array<int, 4> known_bases = {2, 4, 8, 16};
749 if (std::ranges::contains(known_bases, base)) {
750 // Try to convert the character to an integer.
751 usize x;
752 if (std::from_chars(&c, &c + 1, x, base).ec == std::errc{}) {
753 // Success! Push the digits onto the bit-vector.
754 auto digits = static_cast<usize>(std::log2(base));
755 auto old_len = size();
756 resize(old_len + digits);
757 for (auto i = 0uz; i < digits; ++i) this->set(old_len + i, (x >> (digits - i - 1)) & 1);
758 }
759 }
760 return *this;
761 }

780 constexpr BitVector& append_hex_digit(char c) {
781 // Try to convert the hex digit to an integer.
782 int x;
783 if (std::from_chars(&c, &c + 1, x, 16).ec == std::errc{}) {
784 // Success! Push the digits onto the bit-vector.
785 auto old_len = size();
786 resize(old_len + 4);
787 for (auto i = 0uz; i < 4; ++i) this->set(old_len + i, (x >> (3 - i)) & 1);
788 }
789 return *this;

863 template<Unsigned Dst = Word>
864 constexpr std::optional<Dst> split_off_unsigned() {
865 // Edge case ...
866 if (size() == 0) return std::nullopt;
867
868 // Perhaps we can safely cast an `Word` into a `Dst` without any loss of information.
869 constexpr usize bits_per_dst = BITS<Dst>;
870 if constexpr (bits_per_dst <= bits_per_word) {
871 // Easy cases when we can safely cast an `Word` into a `Dst` without any loss of information.
872 // 1. Just one word in the store & we know that all unused bits in that word are zeros.
873 // 2. More than one word in the store but the last word is fully occupied.
874 auto n_words = words();
875 auto shift = size() % bits_per_word;
876 if (n_words == 1 || shift == 0) {
877 auto result = static_cast<Dst>(m_store[n_words - 1]);
878 resize(size() - bits_per_dst);
879 return result;
880 }
881
882 // The last word is not fully occupied so we need to combine it with the next to last word to form a
883 // value that can be cast to a `Dst`.
884 auto lo = m_store[n_words - 2] >> shift;
885 auto hi = m_store[n_words - 1] << (bits_per_word - shift);
886 auto result = static_cast<Dst>(lo | hi);
887 resize(size() - bits_per_dst);
888 return result;
889 } else {
890 // `Dst` is larger than `Word` -- it should be an integer multiple of `bits_per_word` is size.
891 static_assert(bits_per_dst % bits_per_word == 0, "sizeof(Dst) % sizeof(Word) != 0");
892
893 // Pop an appropriate number of `Word` sized chunks off the end of the bit-vector.
894 auto words_per_dst = bits_per_dst / bits_per_word;
895 std::vector<Word> chunks(words_per_dst);
896 for (auto i = 0uz; i < words_per_dst; ++i) chunks[i] = *split_off_unsigned<Word>();
897
898 // Combine the chunks into a `Dst` value.
899 Dst result = 0;
900 for (auto i = words_per_dst; i-- > 0;) result |= static_cast<Dst>(chunks[i] << (i * bits_per_word));
901
902 return result;
903 }

70template<typename Store>
71concept BitStore = requires(Store store, const Store const_store) {
73 typename Store::word_type;
74 requires Unsigned<typename Store::word_type>;
75
77 { const_store.size() } -> std::same_as<usize>;
78
80 { const_store.store() } -> std::same_as<const typename Store::word_type*>;
81
84 { store.store() } -> std::convertible_to<const typename Store::word_type*>;
85
88 { const_store.offset() } -> std::same_as<u8>;
89
95 { const_store.words() } -> std::same_as<usize>;
96
108 { const_store.word(usize{}) } -> std::same_as<typename Store::word_type>;
109
122 { store.set_word(usize{}, typename Store::word_type{}) } -> std::same_as<void>;
123};

305template<BitStore Store>
306constexpr void
307swap(Store& store, usize i0, usize i1) {
308 gf2_debug_assert(i0 < store.size(), "index {} is out of bounds for a store of length {}", i0, store.size());
309 gf2_debug_assert(i1 < store.size(), "index {} is out of bounds for a store of length {}", i1, store.size());
310 if (i0 != i1) {
311 auto [w0, mask0] = index_and_mask<typename Store::word_type>(i0);
312 auto [w1, mask1] = index_and_mask<typename Store::word_type>(i1);
313 auto word0 = store.word(w0);
314 auto word1 = store.word(w1);
315 auto val0 = (word0 & mask0) != 0;
316 auto val1 = (word1 & mask1) != 0;
317 if (val0 != val1) {
318 if (w0 == w1) {
319 // Both bits are in the same word
320 store.set_word(w0, word0 ^ mask0 ^ mask1);
321 } else {
322 // The two bits are in different words
323 store.set_word(w0, word0 ^ mask0);
324 store.set_word(w1, word1 ^ mask1);
325 }
326 }
327 }

383template<BitStore Store>
384constexpr bool
385all(Store const& store) {
386 auto num_words = store.words();
387 if (num_words == 0) { return true; }
388
389 // Check the fully occupied words ...
390 using word_type = typename Store::word_type;
391 for (auto i = 0uz; i < num_words - 1; ++i) {
392 if (store.word(i) != MAX<word_type>) return false;
393 }
394
395 // Last word might not be fully occupied ...
396 auto bits_per_word = BITS<word_type>;
397 auto tail_bits = store.size() % bits_per_word;
398 auto unused_bits = tail_bits == 0 ? 0 : bits_per_word - tail_bits;
399 auto last_word_max = MAX<word_type> >> unused_bits;
400 if (store.word(num_words - 1) != last_word_max) return false;
401
402 return true;
403}

483template<Unsigned Src, BitStore Store>
484constexpr void
485copy(Src src, Store& store) {
486 constexpr auto src_bits = BITS<Src>;
487 gf2_always_assert(store.size() == src_bits, "Lengths do not match: {} != {}.", store.size(), src_bits);
488
489 using word_type = typename Store::word_type;
490 constexpr auto bits_per_word = BITS<word_type>;
491 if constexpr (src_bits <= bits_per_word) {
492 // The source fits into a single store word
493 store.set_word(0, static_cast<word_type>(src));
494 } else {
495 // The source spans some number of store words (which we assume is an integer number).
496 constexpr auto num_words = src_bits / bits_per_word;
497 for (auto i = 0uz; i < num_words; ++i) {
498 auto shift = i * bits_per_word;
499 auto word_value = static_cast<word_type>((src >> shift) & MAX<word_type>);
500 store.set_word(i, word_value);
501 }
502 }
503}

522template<BitStore Src, BitStore Store>
523constexpr void
524copy(Src const& src, Store& store) {
525 // The sizes must match ...
526 gf2_always_assert(store.size() == src.size(), "Lengths do not match: {} != {}.", store.size(), src.size());
527
528 // What is the word types (without any const qualifiers)?
529 using word_type = std::remove_const_t<typename Store::word_type>;
530 using src_word_type = std::remove_const_t<typename Src::word_type>;
531
532 // Numbers of bits per respective words
533 constexpr auto word_bits = BITS<word_type>;
534 constexpr auto src_word_bits = BITS<src_word_type>;
535
536 // What we do next depends on the size of the respective words.
537 if constexpr (word_bits == src_word_bits) {
538 // Easiest case: Same sized words -- we can copy whole words directly ...
539 for (auto i = 0uz; i < store.words(); ++i) {
540 auto src_word = static_cast<word_type>(src.word(i));
541 store.set_word(i, src_word);
542 }
543 } else if constexpr (word_bits > src_word_bits) {
544 // Our word type is larger -- some number of source words fit into each store word.
545 // We assume that our word size is an integer multiple of the source word size.
546 constexpr auto ratio = word_bits / src_word_bits;
547
548 auto src_words = src.words();
549 for (auto i = 0uz; i < store.words(); ++i) {
550
551 // Combine `ratio` source words into a single destination word being careful not to run off the end ...
552 word_type dst_word = 0;
553 for (auto j = 0uz; j < ratio; ++j) {
554 auto src_index = i * ratio + j;
555 word_type src_word = 0;
556 if (src_index < src_words) src_word = static_cast<word_type>(src.word(src_index));
557 dst_word |= (src_word << (j * src_word_bits));
558 }
559 store.set_word(i, dst_word);
560 }
561 } else {
562 // The store word size is smaller -- each source word becomes multiple destination words.
563 // We assume that the source word size is an integer multiple of our word size.
564 constexpr auto ratio = src_word_bits / word_bits;
565
566 auto dst_words = store.words();
567 for (auto src_index = 0uz; src_index < src.words(); ++src_index) {
568 auto src_word = src.word(src_index);
569
570 // Split the source word into `ratio` destination words ...
571 for (auto j = 0uz; j < ratio; ++j) {
572 auto dst_index = src_index * ratio + j;
573 if (dst_index >= dst_words) break;
574 auto shift = j * word_bits;
575 auto dst_word = static_cast<word_type>((src_word >> shift) & MAX<word_type>);
576 store.set_word(dst_index, dst_word);
577 }
578 }
579 }

643template<BitStore Store>
644constexpr void
645fill_random(Store& store, double p = 0.5, u64 seed = 0) {
646 // Keep a single static RNG per thread for all calls to this method, seeded with entropy on the first call.
647 thread_local RNG rng;
648
649 // Edge case handling ...
650 if (p < 0) {
651 set_all(store, false);
652 return;
653 }
654
655 // Scale p by 2^64 to remove floating point arithmetic from the main loop below.
656 // If we determine p rounds to 1 then we can just set all elements to 1 and return early.
657 p = p * 0x1p64 + 0.5;
658 if (p >= 0x1p64) {
659 set_all(store);
660 return;
661 }
662
663 // p does not round to 1 so we use a 64-bit URNG and check each draw against the 64-bit scaled p.
664 auto scaled_p = static_cast<u64>(p);
665
666 // If a seed was provided, set the RNG's seed to it. Otherwise, we carry on from where we left off.
667 u64 old_seed = rng.seed();
668 if (seed != 0) rng.set_seed(seed);
669
670 set_all(store, false);
671 for (auto i = 0uz; i < store.size(); ++i)
672 if (rng.u64() < scaled_p) set(store, i);
673
674 // Restore the old seed if necessary.
675 if (seed != 0) rng.set_seed(old_seed);
676}

725template<BitStore Store>
726constexpr usize
727leading_zeros(Store const& store) {
728 // Note: Even if the last word is not fully occupied, we know unused bits are set to 0.
729 auto bits_per_word = BITS<typename Store::word_type>;
730 for (auto i = 0uz; i < store.words(); ++i) {
731 auto w = store.word(i);
732 if (w != 0) return i * bits_per_word + static_cast<usize>(gf2::trailing_zeros(w));
733 }
734 return store.size();
735}

746template<BitStore Store>
747constexpr usize
748trailing_zeros(Store const& store) {
749 if (is_empty(store)) return 0;
750
751 // The last word may not be fully occupied so we need to subtract the number of unused bits.
752 auto bits_per_word = BITS<typename Store::word_type>;
753 auto tail_bits = store.size() % bits_per_word;
754 auto unused_bits = tail_bits == 0 ? 0 : bits_per_word - tail_bits;
755 auto num_words = store.words();
756 auto i = num_words;
757 while (i--) {
758 auto w = store.word(i);
759 if (w != 0) {
760 auto whole_count = (num_words - i - 1) * bits_per_word;
761 auto partial_count = static_cast<usize>(gf2::leading_zeros(w)) - unused_bits;
762 return whole_count + partial_count;
763 }
764 }
765 return store.size();
766}

836template<BitStore Store>
837constexpr std::optional<usize>
838next_set(Store const& store, usize index) {
839 // Start our search at index + 1.
840 index = index + 1;
841
842 // Perhaps we are off the end? (This also handles the case of an empty type).
843 if (index >= store.size()) return {};
844
845 // Where is that starting index located in the word store?
846 using word_type = typename Store::word_type;
847 auto bits_per_word = BITS<word_type>;
848 auto [word_index, bit] = index_and_offset<word_type>(index);
849
850 // Iterate forward looking for a word with a new set bit and use the lowest one ...
851 for (auto i = word_index; i < store.words(); ++i) {
852 auto w = store.word(i);
853 if (i == word_index) {
854 // First word -- turn all the bits before our starting bit into zeros so we don't see them as set.
855 reset_bits(w, 0, bit);
856 }
857 if (auto loc = lowest_set_bit(w)) return i * bits_per_word + loc.value();
858 }
859 return {};
860}

1198template<BitStore Store>
1199static constexpr auto
1200span(Store const& store, usize begin, usize end) {
1201 // Check that the range is correctly formed & does not extend beyond the end of the bit-store.
1202 gf2_assert(begin <= end, "Span range [{}, {}) is mis-ordered.", begin, end);
1203 gf2_assert(end <= store.size(), "Span end {} extends beyond the store end {}.", end, store.size());
1204
1205 // Get the zero-based word index and bit offset in that word for the begin location.
1206 using word_type = typename Store::word_type;
1207 auto bits_per_word = BITS<word_type>;
1208 auto [data_index, bit_offset] = index_and_offset<word_type>(begin);
1209
1210 // If the store is already a span, we need to adjust those values.
1211 bit_offset += store.offset();
1212 if (bit_offset >= bits_per_word) {
1213 data_index += 1;
1214 bit_offset %= bits_per_word;
1215 }
1216
1217 // Return the read-only bit-span over the underlying word data.
1218 return BitSpan<const word_type>(store.store() + data_index, bit_offset, end - begin);
1219}

1389template<BitStore Store>
1390constexpr void
1391riffle(Store const& store, BitVector<typename Store::word_type>& dst) {
1392 // Nothing to do if the store is empty or has just one element. With two bits `ab` we fill `dst` with `a0b`.
1393 auto sz = store.size();
1394 if (sz < 2) {
1395 dst.resize(sz);
1396 dst.copy(store);
1397 return;
1398 }
1399
1400 // Make sure `dst` is large enough to hold the riffled bits (a bit too big but we fix that below).
1401 dst.resize(2 * sz);
1402 auto dst_words = dst.words();
1403
1404 // Riffle each word in the store into two adjacent words in `dst`.
1405 for (auto i = 0uz; i < store.words(); ++i) {
1406 auto [lo, hi] = riffle(store.word(i));
1407 dst.set_word(2 * i, lo);
1408
1409 // Note that `hi` may be completely superfluous ...
1410 if (2 * i + 1 < dst_words) dst.set_word(2 * i + 1, hi);
1411 }
1412
1413 // If this bit-store was say `abcde` then `dst` will now be `a0b0c0d0e0`. Pop the last 0.
1414 dst.pop();
1415}

1458template<BitStore Store>
1459static std::string
1460to_binary_string(Store const& store, std::string_view sep = "", std::string_view pre = "", std::string_view post = "") {
1461 // Edge case ...
1462 if (is_empty(store)) { return std::format("{}{}", pre, post); }
1463
1464 // Start with the raw binary string by iterating over the words ...
1465 // We preallocate enough space to hold the entire string (actually this is usually a bit bigger than we need).
1466 auto num_words = store.words();
1467 std::string binary_string;
1468 binary_string.reserve(num_words * BITS<typename Store::word_type>);
1469 for (auto i = 0uz; i < num_words; ++i) {
1470 auto w = reverse_bits(store.word(i));
1471 binary_string.append(gf2::to_binary_string(w));
1472 }
1473
1474 // The last word may not be fully occupied and we need to remove the spurious zeros
1475 binary_string.resize(store.size());
1476
1477 // If there is no separator, return early ...
1478 if (sep.empty()) { return std::format("{}{}{}", pre, binary_string, post); }
1479
1480 // Build `result` with separators between each character ...
1481 std::string result;
1482 result.reserve(pre.size() + store.size() * (sep.size() + 1) + post.size());
1483 result.append(pre);
1484 for (auto i = 0uz; i < store.size(); ++i) {
1485 result.push_back(binary_string[i]);
1486 if (i != store.size() - 1) { result.append(sep); }
1487 }
1488 result.append(post);
1489 return result;
1490}

1563template<BitStore Store>
1564static std::string
1565to_hex_string(Store const& store) {
1566 // Edge case: No bits in the store, return the empty string.
1567 if (is_empty(store)) return "";
1568
1569 // The number of digits in the output string. Generally hexadecimal but the last may be to a lower base.
1570 auto digits = (store.size() + 3) / 4;
1571
1572 // Preallocate space allowing for a possible lower base on the last digit such as "_2".
1573 std::string result;
1574 result.reserve(digits + 2);
1575
1576 // Reverse the bits in each word to vector-order and get the fully padded hex string representation.
1577 using word_type = typename Store::word_type;
1578 for (auto i = 0uz; i < store.words(); ++i) {
1579 auto w = reverse_bits(store.word(i));
1580 result.append(gf2::to_hex_string<word_type>(w));
1581 }
1582
1583 // Last word may not be fully occupied and padded with spurious zeros so we truncate the output string.
1584 result.resize(digits);
1585
1586 // Every four elements is encoded by a single hex digit but `size()` may not be a multiple of 4.
1587 auto k = store.size() % 4;
1588 if (k != 0) {
1589 // That last hex digit should really be encoded to a lower base -- 2, 4 or 8.
1590 // We compute the number represented by the trailing `k` elements in the bit-vector.
1591 int num = 0;
1592 for (auto i = 0uz; i < k; ++i)
1593 if (get(store, store.size() - 1 - i)) num |= 1 << i;
1594
1595 // Convert that number to hex & use it to *replace* the last hex digit in our `result` string.
1596 // Also append the appropriate base to the output string so that the last digit can be interpreted properly.
1597 result.resize(result.size() - 1);
1598 result.append(std::format("{:X}.{}", num, 1 << k));
1599 }
1600 return result;
1601}

1683template<BitStore Store>
1684constexpr void
1685operator<<=(Store& store, usize shift) {
1686 // Edge cases where there is nothing to do.
1687 if (shift == 0 || store.size() == 0) return;
1688
1689 // Perhaps we are shifting all the bits out and are left with all zeros.
1690 if (shift >= store.size()) {
1691 set_all(store, false);
1692 return;
1693 }
1694
1695 // For larger shifts, we can efficiently shift by whole words first.
1696 using word_type = typename Store::word_type;
1697 auto bits_per_word = BITS<word_type>;
1698 usize word_shift = shift / bits_per_word;
1699 usize end_word = store.words() - word_shift;
1700
1701 // Do the whole word shifts first, pushing in zero words to fill the empty slots.
1702 if (word_shift > 0) {
1703 // Shift whole words -- starting at the beginning of the store.
1704 for (auto i = 0uz; i < end_word; ++i) store.set_word(i, store.word(i + word_shift));
1705
1706 // Fill in the high order words with zeros.
1707 for (auto i = end_word; i < store.words(); ++i) store.set_word(i, 0);
1708
1709 // How many bits are left to shift?
1710 shift -= word_shift * bits_per_word;
1711 }
1712
1713 // Perhaps there are some partial word shifts left to do.
1714 if (shift != 0) {
1715 // Do the "interior" words where the shift moves bits from one word to the next.
1716 if (end_word > 0) {
1717 auto shift_complement = bits_per_word - shift;
1718 for (auto i = 0uz; i < end_word - 1; ++i) {
1719 auto lo = static_cast<word_type>(store.word(i) >> shift);
1720 auto hi = static_cast<word_type>(store.word(i + 1) << shift_complement);
1721 store.set_word(i, lo | hi);
1722 }
1723 }
1724
1725 // Do the last word.
1726 auto value = store.word(end_word - 1);
1727 store.set_word(end_word - 1, static_cast<word_type>(value >> shift));
1728 }
1729}

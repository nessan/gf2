408 static std::optional<BitVec> from_string(std::string_view sv) {
409 // Edge case ...
410 if (sv.empty()) return BitVec<Word>{};
411
412 // Remove any whitespace, commas, single quotes, or underscores characters.
413 std::string s{sv};
414 std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '\'' || c == '_'; });
415 bool no_punctuation = true;
416
417 // Check for a binary prefix.
418 if (s.starts_with("0b")) return from_binary_string(s, no_punctuation);
419
420 // Check for a hex prefix.
421 if (s.starts_with("0x") || s.starts_with("0X")) return from_hex_string(s, no_punctuation);
422
423 // No prefix, but perhaps the string only contains '0' and '1' characters.
424 if (std::ranges::all_of(s, [](char c) { return c == '0' || c == '1'; })) {
425 return BitVec::from_binary_string(s, no_punctuation);
426 }
427
428 // Last gasp -- try hex ...
429 return BitVec::from_hex_string(s, no_punctuation);
430 }

445 static std::optional<BitVec> from_binary_string(std::string_view sv, bool no_punctuation = false) {
446 // Edge case ...
447 if (sv.empty()) return BitVec<Word>{};
448
449 // Convert to a string & remove the "0b" prefix if it exists.
450 std::string s{sv};
451 if (s.starts_with("0b")) s = s.substr(2);
452
453 // If necessary, also remove any whitespace, commas, single quotes, or underscores.
454 if (!no_punctuation) std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '\'' || c == '_'; });
455
456 // The string should now be a sequence of 0's and 1's.
457 if (std::ranges::any_of(s, [](char c) { return c != '0' && c != '1'; })) return std::nullopt;
458
459 // Construct the bit-vector.
460 auto result = BitVec::zeros(s.size());
461 for (auto i = 0uz; i < s.size(); ++i)
462 if (s[i] == '1') result.set(i);
463 return result;
464 }

486 static std::optional<BitVec> from_hex_string(std::string_view sv, bool no_punctuation = false) {
487 // Edge case ...
488 if (sv.empty()) return BitVec<Word>{};
489
490 // Convert to a string and remove the optional "0x" prefix if it exists.
491 std::string s{sv};
492 if (s.starts_with("0x") || s.starts_with("0X")) s = s.substr(2);
493
494 // By default, the base of the last digit is 16 just like all the others.
495 // However, there may be a suffix of the form ".base" where `base` is one of 2, 4 or 8.
496 int last_digit_base = 16;
497 if (s.ends_with(".2"))
498 last_digit_base = 2;
499 else if (s.ends_with(".4"))
500 last_digit_base = 4;
501 else if (s.ends_with(".8"))
502 last_digit_base = 8;
503
504 // Remove the suffix if it exists.
505 if (last_digit_base != 16) s = s.substr(0, s.size() - 2);
506
507 // If necessary, also remove any whitespace, commas, single quotes, or underscores.
508 if (!no_punctuation) std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '_'; });
509
510 // The string should now be a sequence of 0-9, A-F, a-f.
511 if (std::ranges::any_of(s, [](char c) { return !std::isxdigit(c); })) return std::nullopt;
512
513 // Construct the bit-vector where we allow all the characters to be hex digits.
514 auto result = BitVec::with_capacity(s.size() * 4);
515
516 // Push all but the last character -- those are hex digits for sure.
517 for (auto i = 0uz; i < s.size() - 1; ++i) result.append_hex_digit(s[i]);
518
519 // Push the last character -- it may be a hex digit or a base indicator.
520 result.append_digit(s[s.size() - 1], last_digit_base);
521
522 return result;
523 }

591 constexpr BitVec& resize(usize n) {
592 if (n != size()) {
593 m_store.resize(gf2::words_needed<Word>(n), 0);
594 auto old_size = size();
595 m_size = n;

597 // If we truncated, we need to clean up the last occupied word if necessary.
598 if (n < old_size) clean();

606 constexpr void clean() {
607 // NOTE: This works fine even if `size() == 0`.
608 auto shift = m_size % bits_per_word;
609 if (shift != 0) m_store[words() - 1] &= Word(~(MAX<Word> << shift));
610 }

743 constexpr BitVec& append_digit(char c, int base) {
744 // The known bases are 2, 4, 8, and 16.
745 static constexpr std::array<int, 4> known_bases = {2, 4, 8, 16};
746 if (std::ranges::contains(known_bases, base)) {
747 // Try to convert the character to an integer.
748 usize x;
749 if (std::from_chars(&c, &c + 1, x, base).ec == std::errc{}) {
750 // Success! Push the digits onto the bit-vector.
751 auto digits = static_cast<usize>(std::log2(base));
752 auto old_len = size();
753 resize(old_len + digits);
754 for (auto i = 0uz; i < digits; ++i) this->set(old_len + i, (x >> (digits - i - 1)) & 1);
755 }
756 }
757 return *this;
758 }

777 constexpr BitVec& append_hex_digit(char c) {
778 // Try to convert the hex digit to an integer.
779 int x;
780 if (std::from_chars(&c, &c + 1, x, 16).ec == std::errc{}) {
781 // Success! Push the digits onto the bit-vector.
782 auto old_len = size();
783 resize(old_len + 4);
784 for (auto i = 0uz; i < 4; ++i) this->set(old_len + i, (x >> (3 - i)) & 1);
785 }
786 return *this;

858 template<Unsigned Dst = Word>
859 constexpr std::optional<Dst> split_off_unsigned() {
860 // Edge case ...
861 if (size() == 0) return std::nullopt;
862
863 // Perhaps we can safely cast an `Word` into a `Dst` without any loss of information.
864 constexpr usize bits_per_dst = BITS<Dst>;
865 if constexpr (bits_per_dst <= bits_per_word) {
866 // Easy cases when we can safely cast an `Word` into a `Dst` without any loss of information.
867 // 1. Just one word in the store & we know that all unused bits in that word are zeros.
868 // 2. More than one word in the store but the last word is fully occupied.
869 auto n_words = words();
870 auto shift = size() % bits_per_word;
871 if (n_words == 1 || shift == 0) {
872 auto result = static_cast<Dst>(m_store[n_words - 1]);
873 resize(size() - bits_per_dst);
874 return result;
875 }
876
877 // The last word is not fully occupied so we need to combine it with the next to last word to form a
878 // value that can be cast to a `Dst`.
879 auto lo = m_store[n_words - 2] >> shift;
880 auto hi = m_store[n_words - 1] << (bits_per_word - shift);
881 auto result = static_cast<Dst>(lo | hi);
882 resize(size() - bits_per_dst);
883 return result;
884 } else {
885 // `Dst` is larger than `Word` -- it should be an integer multiple of `bits_per_word` is size.
886 static_assert(bits_per_dst % bits_per_word == 0, "sizeof(Dst) % sizeof(Word) != 0");
887
888 // Pop an appropriate number of `Word` sized chunks off the end of the bit-vector.
889 auto words_per_dst = bits_per_dst / bits_per_word;
890 std::vector<Word> chunks(words_per_dst);
891 for (auto i = 0uz; i < words_per_dst; ++i) chunks[i] = *split_off_unsigned<Word>();
892
893 // Combine the chunks into a `Dst` value.
894 Dst result = 0;
895 for (auto i = words_per_dst; i-- > 0;) result |= static_cast<Dst>(chunks[i] << (i * bits_per_word));
896
897 return result;
898 }
899 }

427 constexpr BitPolynomial& operator+=(BitPolynomial const& rhs) {
428 // Edge case.
429 if (rhs.is_zero()) return *this;
430
431 // Another edge case.
432 if (is_zero()) {
433 *this = rhs;
434 return *this;
435 }
436
437 // Perhaps we need to get bigger to accommodate the rhs? Note that added coefficients are zeros.
438 auto rhs_degree = rhs.degree();
439 if (m_coeffs.size() < rhs_degree + 1) m_coeffs.resize(rhs_degree + 1);
440
441 // Add in the active rhs words.
442 for (auto i = 0uz; i < rhs.monic_word_count(); ++i) {
443 m_coeffs.set_word(i, m_coeffs.word(i) ^ rhs.m_coeffs.word(i));
444 }
445 return *this;
446 }

718 constexpr bool operator()(bool x) const {
719 // Edge case: the zero BitPolynomial.
720 if (is_zero()) { return false; }
721
722 // Edge case: `x = false` which is the same as `x = 0` & we always have p(0) = p_0.
723 if (!x) { return m_coeffs.get(0); }
724
725 // We are evaluating the BitPolynomial at `x = true` which is the same as `x = 1`: p(1) = p_0 + p_1 + p_2 + ...
726 Word sum = 0;
727 for (auto i = 0uz; i < m_coeffs.words(); ++i) sum ^= m_coeffs.word(i);
728 return gf2::count_ones(sum) % 2 == 1;
729 }

746 constexpr auto operator()(BitMatrix<Word> const& M) const {
747 // The bit-matrix argument must be square.
748 gf2_assert(M.is_square(), "Matrix must be square -- not {} x {}!", M.rows(), M.cols());
749
750 // The returned bit-matrix will be n x n.
751 auto n = M.rows();
752
753 // Edge case: If the polynomial is zero then the return value is the n x n zero matrix.
754 if (is_zero()) return BitMatrix<Word>{n, n};
755
756 // Otherwise we start with the polynomial sum being the n x n identity matrix.
757 auto result = BitMatrix<Word>::identity(n);
758
759 // Work backwards a la Horner ...
760 auto d = degree();
761 while (d > 0) {
762
763 // Always multiply ...
764 result = dot(M, result);
765
766 // Add the identity to the sum if the corresponding polynomial coefficient is 1.
767 if (m_coeffs[d - 1]) result.add_identity();
768
769 // And count down ...
770 d--;
771 }
772
773 return result;
774 }

798 BitPolynomial reduce_x_to_the(usize n, bool n_is_log2 = false) const {
799 // Error check: anything mod 0 is not defined.
800 if (is_zero()) throw std::invalid_argument("... mod P(x) is not defined for P(x) := 0.");
801
802 // Edge case: anything mod 1 = 0.
803 if (is_one()) return BitPolynomial<Word>::zero();
804
805 // Edge case: x^0 = 1 and 1 mod P(x) = 1 for any P(x) != 1 (we already handled the case where P(x) := 1).
806 if (n == 0 && !n_is_log2) return BitPolynomial<Word>::one();
807
808 // The BitPolynomial P(x) is non-zero so can be written as P(x) = x^d + p(x) where degree(p) < d.
809 auto d = degree();
810
811 // Edge case: P(x) = x + c where c is a constant so x = P(x) + c (subtraction in GF(2) is the same as addition).
812 // Then x^e = (P(x) + c)^e = terms in powers of P(x) + c^e. Hence, x^e mod P(x) = c^e = c.
813 if (d == 1) return BitPolynomial<Word>::constant(m_coeffs.get(1));
814
815 // We can write p(x) = p_0 + p_1 x + ... + p_{d-1} x^{d-1}. All that matters are those coefficients.
816 auto p = m_coeffs.sub(0, d);
817
818 // A lambda to perform: q(x) <- x*q(x) mod P(x) where degree(q) < d.
819 // The lambda works on the coefficients of q(x) passed as a bit-vector q.
820 auto times_x_step = [&](auto& q) {
821 bool add_p = q[d - 1];
822 q >>= 1;
823 if (add_p) q ^= p;
824 };
825
826 // Iteratively precompute x^{d+i} mod P(x) for i = 0, 1, ..., d-1 starting with x^d mod P(x) ~ p.
827 // We store all the bit-vectors in a standard `std::vector` of length d.
828 std::vector<coeffs_type> power_mod(d, coeffs_type{d});
829 power_mod[0] = p;
830 for (auto i = 1uz; i < d; ++i) {
831 power_mod[i] = power_mod[i - 1];
832 times_x_step(power_mod[i]);
833 }
834
835 // Some workspace we use/reuse below in order to minimize allocations/deallocations.
836 coeffs_type s{2 * d}, h{d};
837
838 // lambda to perform: q(x) <- q(x)^2 mod P(x) where degree(q) < d.
839 // The lambda works on the coefficients of q(x) passed as a bit-vector q.
840 auto square_step = [&](auto& q) {
841 // Square q(x) storing the result in workspace `s`.
842 q.riffled(s);
843
844 // Split s(x) = l(x) + x^d * h(x) where l(x) & h(x) are both of degree less than d.
845 // We reuse q(x) for l(x).
846 s.split_at(d, q, h);
847
848 // s(x) = q(x) + h(x) so s(x) mod P(x) = q(x) + h(x) mod P(x) which we handle term by term.
849 // If h(x) != 0 then at most every second term in h(x) is 1 (nature of BitPolynomial squares in GF(2)).
850 if (auto h_first = h.first_set()) {
851 auto h_last = h.last_set();
852 for (auto i = *h_first; i <= *h_last; i += 2)
853 if (h[i]) q ^= power_mod[i];
854 }
855 };
856
857 // Our return value r(x) := x^e mod P(x) has degree < d: r(x) = r_0 + r_1 x + ... + r_{d-1} x^{d-1}.
858 coeffs_type r{d};
859
860 // If `n_is_log2` is `true`, we are reducing x^(2^n) mod P(x) which is done iteratively by squaring.
861 if (n_is_log2) {
862 // Note that we already handled edge case where P(x) = x + c above.
863 // Start with r(x) = x mod P(x) -> x^2 mod P(x) -> x^4 mod P(x) ...
864 r[1] = true;
865 for (auto i = 0uz; i < n; ++i) square_step(r);
866 return BitPolynomial{std::move(r)};
867 }
868
869 // Small exponent case: n < d => x^n mod P(x) = x^n.
870 if (n < d) return BitPolynomial::x_to_the(n);
871
872 // Matching exponent case: n = d => x^n mod P(x) = x^d mod P(x) = p(x).
873 if (n == d) return BitPolynomial{std::move(p)};
874
875 // General case: n > d is handled by a square & multiply algorithm.
876 usize n_bit = std::bit_floor(n);
877
878 // Start with r(x) = x mod P(x) which takes care of the most significant binary digit in n.
879 // TODO: We could start a bit further along with a higher power r(x) := x^? mod P(x) where ? < n but > 1.
880 r[1] = 1;
881 n_bit >>= 1;
882
883 // And off we go ...
884 while (n_bit) {
885
886 // Always do a square step and then a times_x step if necessary (i.e. if current bit in N is set).
887 square_step(r);
888 if (n & n_bit) times_x_step(r);
889
890 // On to the next bit position in n.
891 n_bit >>= 1;
892 }
893
894 // Made it.
895 return BitPolynomial{std::move(r)};
896 }

914 std::string to_string(std::string_view var = "x") const {
915 // Edge case: the zero BitPolynomial.
916 if (is_zero()) return "0";
917
918 // Otherwise we construct the string ...
919 std::ostringstream ss;
920
921 // All terms other than the first one are preceded by a "+"
922 bool first_term = true;
923 for (auto i = 0uz; i <= degree(); ++i) {
924 if (m_coeffs.get(i)) {
925 if (i == 0) {
926 ss << "1";
927 } else {
928 if (!first_term) ss << " + ";
929 if (i == 1)
930 ss << var;
931 else
932 ss << var << "^" << i;
933 }
934 first_term = false;
935 }
936 }
937 return ss.str();
938 }

952 std::string to_full_string(std::string_view var = "x") const {
953 // Edge case: the zero BitPolynomial.
954 if (is_empty()) return "0";
955
956 // Otherwise we construct the string ...
957 std::ostringstream ss;
958
959 // Start with the first term ...
960 if (m_coeffs.get(0))
961 ss << "1";
962 else
963 ss << "0";
964
965 // Now all the other terms, each preceded by a "+"
966 for (auto i = 1uz; i < size(); ++i) {
967 ss << " + ";
968 if (m_coeffs.get(i) == false) ss << "0";
969 if (i == 1)
970 ss << var;
971 else
972 ss << var << "^" << i;
973 }
974 return ss.str();
975 }

995 friend constexpr bool operator==(BitPolynomial const& lhs, BitPolynomial const& rhs) {
996 // Edge case.
997 if (&lhs == &rhs) return true;
998
999 // Check the active words for equality.
1000 auto count = lhs.monic_word_count();
1001 if (rhs.monic_word_count() != count) return false;
1002 for (auto i = 0uz; i < count; ++i)
1003 if (lhs.m_coeffs.word(i) != rhs.m_coeffs.word(i)) return false;
1004
1005 // Made it
1006 return true;
1007 }

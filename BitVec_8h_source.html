384 static std::optional<BitVec> from_string(std::string_view sv) {
385 // Edge case ...
386 if (sv.empty()) return BitVec<Word>{};
387
388 // Remove any whitespace, commas, single quotes, or underscores characters.
389 std::string s{sv};
390 std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '\'' || c == '_'; });
391 bool no_punctuation = true;
392
393 // Check for a binary prefix.
394 if (s.starts_with("0b")) return from_binary_string(s, no_punctuation);
395
396 // Check for a hex prefix.
397 if (s.starts_with("0x") || s.starts_with("0X")) return from_hex_string(s, no_punctuation);
398
399 // No prefix, but perhaps the string only contains '0' and '1' characters.
400 if (std::ranges::all_of(s, [](char c) { return c == '0' || c == '1'; })) {
401 return BitVec::from_binary_string(s, no_punctuation);
402 }
403
404 // Last gasp -- try hex ...
405 return BitVec::from_hex_string(s, no_punctuation);
406 }

421 static std::optional<BitVec> from_binary_string(std::string_view sv, bool no_punctuation = false) {
422 // Edge case ...
423 if (sv.empty()) return BitVec<Word>{};
424
425 // Convert to a string & remove the "0b" prefix if it exists.
426 std::string s{sv};
427 if (s.starts_with("0b")) s = s.substr(2);
428
429 // If necessary, also remove any whitespace, commas, single quotes, or underscores.
430 if (!no_punctuation) std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '\'' || c == '_'; });
431
432 // The string should now be a sequence of 0's and 1's.
433 if (std::ranges::any_of(s, [](char c) { return c != '0' && c != '1'; })) return std::nullopt;
434
435 // Construct the bit-vector.
436 auto result = BitVec::zeros(s.size());
437 for (auto i = 0uz; i < s.size(); ++i)
438 if (s[i] == '1') result.set(i);
439 return result;
440 }

462 static std::optional<BitVec> from_hex_string(std::string_view sv, bool no_punctuation = false) {
463 // Edge case ...
464 if (sv.empty()) return BitVec<Word>{};
465
466 // Convert to a string and remove the optional "0x" prefix if it exists.
467 std::string s{sv};
468 if (s.starts_with("0x") || s.starts_with("0X")) s = s.substr(2);
469
470 // By default, the base of the last digit is 16 just like all the others.
471 // However, there may be a suffix of the form ".base" where `base` is one of 2, 4 or 8.
472 int last_digit_base = 16;
473 if (s.ends_with(".2"))
474 last_digit_base = 2;
475 else if (s.ends_with(".4"))
476 last_digit_base = 4;
477 else if (s.ends_with(".8"))
478 last_digit_base = 8;
479
480 // Remove the suffix if it exists.
481 if (last_digit_base != 16) s = s.substr(0, s.size() - 2);
482
483 // If necessary, also remove any whitespace, commas, single quotes, or underscores.
484 if (!no_punctuation) std::erase_if(s, [](char c) { return c == ' ' || c == ',' || c == '_'; });
485
486 // The string should now be a sequence of 0-9, A-F, a-f.
487 if (std::ranges::any_of(s, [](char c) { return !std::isxdigit(c); })) return std::nullopt;
488
489 // Construct the bit-vector where we allow all the characters to be hex digits.
490 auto result = BitVec::with_capacity(s.size() * 4);
491
492 // Push all but the last character -- those are hex digits for sure.
493 for (auto i = 0uz; i < s.size() - 1; ++i) result.append_hex_digit(s[i]);
494
495 // Push the last character -- it may be a hex digit or a base indicator.
496 result.append_digit(s[s.size() - 1], last_digit_base);
497
498 return result;
499 }

558 constexpr BitVec& resize(usize n) {
559 if (n != size()) {
560 m_data.resize(gf2::words_needed<Word>(n), 0);
561 auto old_size = size();
562 m_size = n;

564 // If we truncated, we need to clean up the last occupied word if necessary.
565 if (n < old_size) clean();
566 }

573 constexpr void clean() {
574 // NOTE: This works fine even if `size() == 0`.
575 auto shift = m_size % bits_per_word;
576 if (shift != 0) m_data[words() - 1] &= Word(~(MAX<Word> << shift));
577 }

682 constexpr BitVec& append_digit(char c, int base) {
683 // The known bases are 2, 4, 8, and 16.
684 static constexpr std::array<int, 4> known_bases = {2, 4, 8, 16};
685 if (std::ranges::contains(known_bases, base)) {
686 // Try to convert the character to an integer.
687 usize x;
688 if (std::from_chars(&c, &c + 1, x, base).ec == std::errc{}) {
689 // Success! Push the digits onto the bit-vector.
690 auto digits = static_cast<usize>(std::log2(base));
691 auto old_len = size();
692 resize(old_len + digits);
693 for (auto i = 0uz; i < digits; ++i) this->set(old_len + i, (x >> (digits - i - 1)) & 1);
694 }
695 }
696 return *this;
697 }

716 constexpr BitVec& append_hex_digit(char c) {
717 // Try to convert the hex digit to an integer.
718 int x;
719 if (std::from_chars(&c, &c + 1, x, 16).ec == std::errc{}) {
720 // Success! Push the digits onto the bit-vector.
721 auto old_len = size();
722 resize(old_len + 4);
723 for (auto i = 0uz; i < 4; ++i) this->set(old_len + i, (x >> (3 - i)) & 1);
724 }
725 return *this;

821 template<Unsigned Dst = Word>
822 constexpr std::optional<Dst> split_off_unsigned() {
823 // Edge case ...
824 if (size() == 0) return std::nullopt;
825
826 // Perhaps we can safely cast an `Word` into a `Dst` without any loss of information.
827 constexpr usize bits_per_dst = BITS<Dst>;
828 if constexpr (bits_per_dst <= bits_per_word) {
829 // Easy cases when we can safely cast an `Word` into a `Dst` without any loss of information.
830 // 1. Just one word in the store & we know that all unused bits in that word are zeros.
831 // 2. More than one word in the store but the last word is fully occupied.
832 auto n_words = words();
833 auto shift = size() % bits_per_word;
834 if (n_words == 1 || shift == 0) {
835 auto result = static_cast<Dst>(m_data[n_words - 1]);
836 resize(size() - bits_per_dst);
837 return result;
838 }
839
840 // The last word is not fully occupied so we need to combine it with the next to last word to form a
841 // value that can be cast to a `Dst`.
842 auto lo = m_data[n_words - 2] >> shift;
843 auto hi = m_data[n_words - 1] << (bits_per_word - shift);
844 auto result = static_cast<Dst>(lo | hi);
845 resize(size() - bits_per_dst);
846 return result;
847 } else {
848 // `Dst` is larger than `Word` -- it should be an integer multiple of `bits_per_word` is size.
849 static_assert(bits_per_dst % bits_per_word == 0, "sizeof(Dst) % sizeof(Word) != 0");
850
851 // Pop an appropriate number of `Word` sized chunks off the end of the bit-vector.
852 auto words_per_dst = bits_per_dst / bits_per_word;
853 std::vector<Word> chunks(words_per_dst);
854 for (auto i = 0uz; i < words_per_dst; ++i) chunks[i] = *split_off_unsigned<Word>();
855
856 // Combine the chunks into a `Dst` value.
857 Dst result = 0;
858 for (auto i = words_per_dst; i-- > 0;) result |= static_cast<Dst>(chunks[i] << (i * bits_per_word));
859
860 return result;
861 }
862 }

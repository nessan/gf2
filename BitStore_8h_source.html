256 constexpr auto swap(usize i0, usize i1) {
257 gf2_debug_assert(i0 < size(), "index {} is out of bounds for a store of length {}", i0, size());
258 gf2_debug_assert(i1 < size(), "index {} is out of bounds for a store of length {}", i1, size());
259 if (i0 != i1) {
260 auto [w0, mask0] = index_and_mask<word_type>(i0);
261 auto [w1, mask1] = index_and_mask<word_type>(i1);
262 auto word0 = word(w0);
263 auto word1 = word(w1);
264 auto val0 = (word0 & mask0) != 0;
265 auto val1 = (word1 & mask1) != 0;
266 if (val0 != val1) {
267 if (w0 == w1) {
268 // Both bits are in the same word
269 set_word(w0, word0 ^ mask0 ^ mask1);
270 } else {
271 // BitsIter are in different words
272 set_word(w0, word0 ^ mask0);
273 set_word(w1, word1 ^ mask1);
274 }
275 }
276 }
277 return *this;
278 }

325 constexpr bool all() const {
326 auto num_words = words();
327 if (num_words == 0) { return true; }
328
329 // Check the fully occupied words ...
330 for (auto i = 0uz; i < num_words - 1; ++i) {
331 if (word(i) != MAX<word_type>) return false;
332 }
333
334 // Last word might not be fully occupied ...
335 auto unused_bits = bits_per_word - (size() % bits_per_word);
336 auto last_word_max = MAX<word_type> >> unused_bits;
337 if (word(num_words - 1) != last_word_max) return false;
338
339 return true;
340 }

409 template<unsigned_word Src>
410 auto copy(Src src) {
411 constexpr auto src_bits = BITS<Src>;
412 gf2_assert(size() == src_bits, "Lengths do not match: {} != {}.", size(), src_bits);
413
414 if constexpr (src_bits <= bits_per_word) {
415 // The source fits into a single one of our words
416 set_word(0, static_cast<word_type>(src));
417 } else {
418 // The source spans some number of our words (which we assume is an integer number).
419 constexpr auto num_words = src_bits / bits_per_word;
420 for (auto i = 0uz; i < num_words; ++i) {
421 auto shift = i * bits_per_word;
422 auto word_value = static_cast<word_type>((src >> shift) & MAX<word_type>);
423 set_word(i, word_value);
424 }
425 }
426 return *this;
427 }

443 template<typename SrcStore>
444 auto copy(const BitStore<SrcStore>& src) {
445 // The sizes must match ...
446 gf2_assert(size() == src.size(), "Lengths do not match: {} != {}.", size(), src.size());
447
448 // What is the source word type (without any const qualifier)?
449 using src_word_type = std::remove_const_t<typename store_word<SrcStore>::word_type>;
450
451 // Numbers of bits per respective words
452 constexpr auto word_bits = BITS<word_type>;
453 constexpr auto src_word_bits = BITS<src_word_type>;
454
455 // What we do next depends on the size of the respective words.
456 if constexpr (word_bits == src_word_bits) {
457 // Easiest case: Same sized words -- we can copy whole words directly ...
458 for (auto i = 0uz; i < words(); ++i) {
459 auto src_word = static_cast<word_type>(src.word(i));
460 set_word(i, src_word);
461 }
462 } else if constexpr (word_bits > src_word_bits) {
463 // Our word type is larger -- some number of source words fit into each one of ours.
464 // We assume that our word size is an integer multiple of the source word size.
465 constexpr auto ratio = word_bits / src_word_bits;
466
467 auto src_words = src.words();
468 for (auto i = 0uz; i < words(); ++i) {
469
470 // Combine `ratio` source words into a single destination word being careful not to run off the end ...
471 word_type dst_word = 0;
472 for (auto j = 0uz; j < ratio; ++j) {
473 auto src_index = i * ratio + j;
474 word_type src_word = 0;
475 if (src_index < src_words) src_word = static_cast<word_type>(src.word(src_index));
476 dst_word |= (src_word << (j * src_word_bits));
477 }
478 set_word(i, dst_word);
479 }
480 } else {
481 // Our word size is smaller -- each source word becomes multiple destination words.
482 // We assume that the source word size is an integer multiple of our word size.
483 constexpr auto ratio = src_word_bits / word_bits;
484
485 auto dst_words = words();
486 for (auto src_index = 0uz; src_index < src.words(); ++src_index) {
487 auto src_word = src.word(src_index);
488
489 // Split the source word into `ratio` destination words ...
490 for (auto j = 0uz; j < ratio; ++j) {
491 auto dst_index = src_index * ratio + j;
492 if (dst_index >= dst_words) break;
493 auto shift = j * word_bits;
494 auto dst_word = static_cast<word_type>((src_word >> shift) & MAX<word_type>);
495 set_word(dst_index, dst_word);
496 }
497 }
498 }
499 return *this;
500 }

558 auto random_fill(double p = 0.5, u64 seed = 0) {
559 // Keep a single static RNG per thread for all calls to this method, seeded with entropy on the first call.
560 thread_local RNG rng;
561
562 // Edge case handling ...
563 if (p < 0) return set_all(false);
564
565 // Scale p by 2^64 to remove floating point arithmetic from the main loop below.
566 // If we determine p rounds to 1 then we can just set all elements to 1 and return early.
567 p = p * 0x1p64 + 0.5;
568 if (p >= 0x1p64) return set_all();
569
570 // p does not round to 1 so we use a 64-bit URNG and check each draw against the 64-bit scaled p.
571 auto scaled_p = static_cast<u64>(p);
572
573 // If a seed was provided, set the RNG's seed to it. Otherwise, we carry on from where we left off.
574 u64 old_seed = rng.seed();
575 if (seed != 0) rng.set_seed(seed);
576
577 set_all(false);
578 for (auto i = 0uz; i < size(); ++i)
579 if (rng.u64() < scaled_p) set(i);
580
581 // Restore the old seed if necessary.
582 if (seed != 0) rng.set_seed(old_seed);
583 return *this;
584 }

627 constexpr usize leading_zeros() const {
628 // Note: Even if the last word is not fully occupied, we know unused bits are set to 0.
629 for (auto i = 0uz; i < words(); ++i) {
630 auto w = word(i);
631 if (w != 0) return i * bits_per_word + static_cast<usize>(gf2::trailing_zeros(w));
632 }
633 return size();
634 }
635
645 constexpr usize trailing_zeros() const {
646 if (is_empty()) return 0;
647
648 // The last word may not be fully occupied so we need to subtract the number of unused bits.
649 auto unused_bits = bits_per_word - (size() % bits_per_word);
650 auto num_words = words();
651 auto i = num_words;
652 while (i--) {
653 auto w = word(i);
654 if (w != 0) {
655 auto whole_count = (num_words - i - 1) * bits_per_word;
656 auto partial_count = static_cast<usize>(gf2::leading_zeros(w)) - unused_bits;
657 return whole_count + partial_count;
658 }
659 }
660 return size();
661 }

725 std::optional<usize> next_set(usize index) const {
726 // Start our search at index + 1.
727 index = index + 1;
728
729 // Perhaps we are off the end? (This also handles the case of an empty type).
730 if (index >= size()) return {};
731
732 // Where is that starting index located in the word store?
733 auto [word_index, bit] = index_and_offset<word_type>(index);
734
735 // Iterate forward looking for a word with a new set bit and use the lowest one ...
736 for (auto i = word_index; i < words(); ++i) {
737 auto w = word(i);
738 if (i == word_index) {
739 // First word -- turn all the bits before our starting bit into zeros so we don't see them as set.
740 reset_bits(w, 0, bit);
741 }
742 if (auto loc = lowest_set_bit(w)) return i * bits_per_word + loc.value();
743 }
744 return {};
745 }

759 std::optional<usize> previous_set(usize index) const {
760 // Edge case: If the store is empty or we are at the start, there are no previous set bits.
761 if (is_empty() || index == 0) return {};
762
763 // Silently fix large indices and also adjust the index down a slot.
764 if (index > size()) index = size();
765 index--;
766
767 // Where is that starting index located in the word store?
768 auto [word_index, bit] = index_and_offset<word_type>(index);
769
770 // Iterate backwards looking for a word with a new set bit and use the highest one ...
771 for (auto i = word_index + 1; i--;) {
772 auto w = word(i);
773 if (i == word_index) {
774 // First word -- turn all higher bits after our starting bit into zeros so we don't see them as set.
775 reset_bits(w, bit + 1, bits_per_word);
776 }
777 if (auto loc = highest_set_bit(w)) return i * bits_per_word + loc.value();
778 }

1033 constexpr auto span(usize begin, usize end) const {
1034 // Check that the range is correctly formed and non-empty.
1035 gf2_assert(begin < end, "bit-span [{}, {}) is invalid.", begin, end);
1036
1037 // Check that the range does not extend beyond the end of the bit-store.
1038 gf2_assert(end <= size(), "bit-span end {} extends beyond the store end {}.", end, size());
1039
1040 // Get the zero-based word index and bit offset in that word for the begin location.
1041 auto [index, bit_offset] = index_and_offset<word_type>(begin);
1042
1043 // If the store is already a span over another store, we need to adjust those values.
1044 bit_offset += offset();
1045 if (bit_offset >= bits_per_word) {
1046 index += 1;
1047 bit_offset = bit_offset % bits_per_word;
1048 }
1049
1050 // Return the bit-span over the underlying word data.
1051 return BitSpan<const word_type>(data() + index, bit_offset, end - begin);
1052 }

1179 constexpr void riffle_into(BitVec<word_type>& dst) const {
1180 // Nothing to do if the bit-vector is empty or has just one element.
1181 // With two bits `ab` we fill `dst` with `a0b`.
1182 if (size() <= 1) {
1183 dst.copy(*this);
1184 return;
1185 }
1186
1187 // Make sure `dst` is large enough to hold the riffled bits (a bit too big but we fix that below).
1188 dst.resize(2 * size());
1189 auto dst_words = dst.words();
1190
1191 // Riffle each word in `this` into two adjacent words in `dst`.
1192 for (auto i = 0uz; i < words(); ++i) {
1193 auto [lo, hi] = riffle(word(i));
1194 dst.set_word(2 * i, lo);
1195
1196 // Note that `hi` may be completely superfluous ...
1197 if (2 * i + 1 < dst_words) dst.set_word(2 * i + 1, hi);
1198 }
1199
1200 // If this bit-store was say `abcde` then `dst` will now be `a0b0c0d0e0`. Pop the last 0.
1201 dst.pop();
1202 }

1239 constexpr void operator<<=(usize shift) {
1240 // Edge cases where there is nothing to do.
1241 if (shift == 0 || size() == 0) return;
1242
1243 // Perhaps we are shifting all the bits out and are left with all zeros.
1244 if (shift >= size()) {
1245 set_all(false);
1246 return;
1247 }
1248
1249 // For larger shifts, we can efficiently shift by whole words first.
1250 usize word_shift = shift / bits_per_word;
1251 usize end_word = words() - word_shift;
1252
1253 // Do the whole word shifts first, pushing in zero words to fill the empty slots.
1254 if (word_shift > 0) {
1255 // Shift whole words -- starting at the beginning of the store.
1256 for (auto i = 0uz; i < end_word; ++i) set_word(i, word(i + word_shift));
1257
1258 // Fill in the high order words with zeros.
1259 for (auto i = end_word; i < words(); ++i) set_word(i, 0);
1260
1261 // How many bits are left to shift?
1262 shift -= word_shift * bits_per_word;
1263 }
1264
1265 // Perhaps there are some partial word shifts left to do.
1266 if (shift != 0) {
1267 // Do the "interior" words where the shift moves bits from one word to the next.
1268 if (end_word > 0) {
1269 auto shift_complement = bits_per_word - shift;
1270 for (auto i = 0uz; i < end_word - 1; ++i) {
1271 auto lo = static_cast<word_type>(word(i) >> shift);
1272 auto hi = static_cast<word_type>(word(i + 1) << shift_complement);
1273 set_word(i, lo | hi);
1274 }
1275 }
1276
1277 // Do the last word.
1278 auto value = word(end_word - 1);
1279 set_word(end_word - 1, static_cast<word_type>(value >> shift));
1280 }
1281 }

1296 constexpr void operator>>=(usize shift) {
1297 // Edge cases where there is nothing to do.
1298 if (shift == 0 || size() == 0) return;
1299
1300 // Perhaps we are shifting all the bits out and are left with all zeros.
1301 if (shift >= size()) {
1302 set_all(false);
1303 return;
1304 }
1305
1306 // For larger shifts, we can efficiently shift by whole words first.
1307 usize word_shift = shift / bits_per_word;
1308
1309 // Do the whole word shifts first, pushing in zero words to fill the empty slots.
1310 if (word_shift > 0) {
1311 // Shift whole words -- starting at the end of the store.
1312 for (auto i = words(); i-- > word_shift;) set_word(i, word(i - word_shift));
1313
1314 // Fill in the low order words with zeros.
1315 for (auto i = 0uz; i < word_shift; ++i) set_word(i, 0);
1316
1317 // How many bits are left to shift?
1318 shift -= word_shift * bits_per_word;
1319 }
1320
1321 // Perhaps there are some partial word shifts left to do.
1322 if (shift != 0) {
1323 // Do the "interior" words where the shift moves bits from one word to the next.
1324 auto shift_complement = bits_per_word - shift;
1325 for (auto i = words(); i-- > word_shift + 1;) {
1326 auto lo = static_cast<word_type>(word(i - 1) >> shift_complement);
1327 auto hi = static_cast<word_type>(word(i) << shift);
1328 set_word(i, lo | hi);
1329 }
1330 }
1331
1332 // Do the "first" word.
1333 auto value = word(word_shift);
1334 set_word(word_shift, static_cast<word_type>(value << shift));
1335 }

1639 std::string to_binary_string(std::string_view sep = "", std::string_view pre = "",
1640 std::string_view post = "") const {
1641 // Edge case ...
1642 if (is_empty()) { return std::format("{}{}", pre, post); }
1643
1644 // Start with the raw binary string by iterating over the words ...
1645 // We preallocate enough space to hold the entire string (actually this is usually a bit bigger than we need).
1646 auto num_words = words();
1647 std::string binary_string;
1648 binary_string.reserve(num_words * bits_per_word);
1649 for (auto i = 0uz; i < num_words; ++i) {
1650 auto w = reverse_bits(word(i));
1651 binary_string.append(gf2::to_binary_string(w));
1652 }
1653
1654 // The last word may not be fully occupied and we need to remove the spurious zeros
1655 binary_string.resize(size());
1656
1657 // If there is no separator, return early ...
1658 if (sep.empty()) { return std::format("{}{}{}", pre, binary_string, post); }
1659
1660 // Build `result` with separators between each character ...
1661 std::string result;
1662 result.reserve(pre.size() + size() * (sep.size() + 1) + post.size());
1663 result.append(pre);
1664 for (auto i = 0uz; i < size(); ++i) {
1665 result.push_back(binary_string[i]);
1666 if (i != size() - 1) { result.append(sep); }
1667 }
1668 result.append(post);
1669 return result;
1670 }

1702 std::string to_hex_string() const {
1703 // Edge case: No bits in the store, return the empty string.
1704 if (is_empty()) return "";
1705
1706 // The number of digits in the output string. Generally hexadecimal but the last may be to a lower base.
1707 auto digits = (size() + 3) / 4;
1708
1709 // Preallocate space allowing for a possible lower base on the last digit such as "_2".
1710 std::string result;
1711 result.reserve(digits + 2);
1712
1713 // Reverse the bits in each word to vector-order and get the fully padded hex string representation.
1714 for (auto i = 0uz; i < words(); ++i) {
1715 auto w = reverse_bits(word(i));
1716 result.append(gf2::to_hex_string<word_type>(w));
1717 }
1718
1719 // Last word may not be fully occupied and padded with spurious zeros so we truncate the output string.
1720 result.resize(digits);
1721
1722 // Every four elements is encoded by a single hex digit but `size()` may not be a multiple of 4.
1723 auto k = size() % 4;
1724 if (k != 0) {
1725 // That last hex digit should really be encoded to a lower base -- 2, 4 or 8.
1726 // We compute the number represented by the trailing `k` elements in the bit-vector.
1727 int num = 0;
1728 for (auto i = 0uz; i < k; ++i)
1729 if (get(size() - 1 - i)) num |= 1 << i;
1730
1731 // Convert that number to hex & use it to *replace* the last hex digit in our `result` string.
1732 // Also append the appropriate base to the output string so that the last digit can be interpreted properly.
1733 result.resize(result.size() - 1);
1734 result.append(std::format("{:X}.{}", num, 1 << k));
1735 }
1736 return result;
1737 }

1875template<typename Lhs, typename Rhs>
1876 requires store_words_match<Lhs, Rhs>
1877auto
1878convolve(const BitStore<Lhs>& lhs, const BitStore<Rhs>& rhs) {
1879 using word_type = typename store_word<Lhs>::word_type;
1880
1881 // Edge case: if either store is empty then the convolution is empty.
1882 if (lhs.is_empty() || rhs.is_empty()) return BitVec<word_type>{};
1883
1884 // Generally the result will have length `self.size() + other.size() - 1` (could be all zeros).
1885 auto result = BitVec<word_type>::zeros(lhs.size() + rhs.size() - 1);
1886
1887 // If either vector is all zeros then the convolution is all zeros.
1888 if (lhs.none() || rhs.none()) return result;
1889
1890 // Only need to consider words in `rhs` up to and including the one holding its final set bit.
1891 // We have already checked that `rhs` is not all zeros so we know there is a last set bit!
1892 auto rhs_words_end = word_index<word_type>(rhs.last_set().value()) + 1;
1893
1894 // Initialize `result` by copying the live words from `rhs`
1895 for (auto i = 0uz; i < rhs_words_end; ++i) result.set_word(i, rhs.word(i));
1896
1897 // Work backwards from the last set bit in `lhs` (which we know exists as we checked `lhs` is not all zeros).
1898 for (auto i = lhs.last_set().value(); i-- > 0;) {
1899 word_type prev = 0;
1900 for (auto j = 0uz; j < result.words(); ++j) {
1901 auto left = static_cast<word_type>(prev >> (BITS<word_type> - 1));
1902 prev = result.word(j);
1903 result.set_word(j, static_cast<word_type>(prev << 1) | left);
1904 }
1905
1906 if (lhs.get(i)) {
1907 for (auto j = 0uz; j < rhs_words_end; ++j) result.set_word(j, result.word(j) ^ rhs.word(j));
1908 }
1909 }
1910 return result;
1911}
